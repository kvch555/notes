/// 对象的回收机制 ///
class Human {

    public $name = '张三';
    public $gender = null;
    
    public function __destruct() {
        echo '死了!
';
    }

}


$a = new Human();
$b = $c = $d = $a;


echo $a->name,'
';
echo $b->name,'
';

$b->name = '李四';
echo $a->name,'
';
echo $b->name,'
';



unset($a);   // $b,$c,$d在指向对象,因此对象不能销毁.

echo '
';

/*
1:死几次
2:死上灰线上,还是死在灰线下
*/
